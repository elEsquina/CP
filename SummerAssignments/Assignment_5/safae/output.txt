#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;

int n, m, g[605][605];
int main() {
    int i, j, k;
    while(scanf("%d %d", &n, &m) == 2) {
        if(n+m == 0)
            break;
        for(i = 0; i < n; i++)
            for(j = 0; j < m; j++)
                scanf("%d", &g[i][j]);
        int ret = 0;
        for(i = 0; i < m; i++) {
            int sum[605] = {};
            for(j = i; j < m; j++) {
                int height = j - i + 1;
                int l = 0, r = 0;
                for(k = 0; k < n; k++) {
                    if(height == 1)
                        sum[k] = 1;
                    else {
                        if(g[k][j] > g[k][j-1])
                            sum[k]++;
                    }
                    r = k;
                    if(r-1 >= l && g[r-1][j] >= g[r][i])
                        l = max(l, k);
                    if(sum[k] != height) l = max(l, k+1);
                    ret = max(ret, (r-l+1)*height);
                }
            }
        }
        printf("%d\n", ret);
    }
    return 0;
}
